	char choice;
	
	cout << "OPERATIONS: " << endl;

	cout << "b. Subtraction" << endl;
	cout << "c. Multiplication" << endl;
	cout << "d. Division" << endl;

	cout << "choice: ";
	cin >> choice;

	// continue here
